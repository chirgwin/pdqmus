1 /**
  2  * Creates an instance of pdqmus.TimeSignature, an abstract time signature object
  3  *
  4  * @constructor
  5  * @this {pdqmus.TimeSignature}
  6  * @param onset {number} Onset time, in seconds
  7  * @param numerator {number} time signature numerator, integer
  8  * @param denominator {number} time signature denominator, integer
  9  * @param onsetMetric {number} Metric onset, in beats
 10  */
 11 pdqmus.TimeSignature = function(onset, numerator, denominator, onsetMetric)
 12 {
 13     this.onset = onset;
 14     this.numerator = numerator;
 15     this.denominator = denominator;
 16     this.onsetMetric = onsetMetric;
 17     this.duration = 0;
 18 }
 19 
 20 pdqmus.TimeSignature.prototype = new Notation;
 21 
 22 
 23 pdqmus.TimeSignature.prototype.clone = function()
 24 {
 25     return new pdqmus.KeySignature(
 26         this.onset,
 27         this.numerator,
 28         this.denominator,
 29         this.onsetMetric
 30         );
 31 };
