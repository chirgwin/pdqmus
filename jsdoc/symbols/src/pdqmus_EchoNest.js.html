1 /**
  2  * Creates an instance of pdqmus.EchoNest
  3  *
  4  * A wrapper for making calls (via asynchronous HTTP requests) to the EchoNest API
  5  * 
  6  * @constructor
  7  * @this {pdqmus.EchoNest}
  8  * @param {string} apiKey Echo Nest API key
  9  * @param {function} loadedCallback function to call when api call is loaded
 10  */
 11 
 12 pdqmus.EchoNest = function(apiKey, loadedCallback)
 13 {
 14     const BASE_URL = "http://developer.echonest.com/api/v4";
 15     const DEFAULT_NUM_ROWS = 15;
 16     const DELIMITER = "/";
 17     this.apiKey = apiKey;
 18     this.loadedCallback;
 19     this.errorCallback;
 20     this.method = "GET";
 21     this.result;
 22     this.useJsonp = false;
 23     var self = this;
 24 
 25     this.call = function(api, method, params, callback, numRows) 
 26     {    
 27         var asynchRequest = new pdqmus.AsynchRequest(callback);
 28         if (self.errorCallback)
 29         {
 30             asynchRequest.errorCallback = self.errorCallback;
 31         }
 32         
 33         var parameters = new Object();
 34         parameters.api_key = this.apiKey;
 35         for (var key in params)
 36         {
 37             parameters[key] = params[key];
 38         }
 39         if (self.useJsonp)
 40         {        
 41             parameters["format"] = "jsonp";
 42             parameters["callback"] = callback;
 43             var url = Api.formUrl(BASE_URL + DELIMITER + api + DELIMITER + method, parameters);
 44             Api.loadJsonp(url);            
 45         }
 46         else
 47         {            
 48             parameters["format"] = "json";        
 49             var url = Api.formUrl(Api.PROXY_URL + BASE_URL + DELIMITER + api + DELIMITER + method, parameters);
 50             console.log(url);
 51             asynchRequest.loadJson(url);
 52         }
 53         function loadedCallback (result)
 54         {				
 55             self.result = result;
 56             self.loadedCallback(result);
 57         }
 58     }    
 59 }
