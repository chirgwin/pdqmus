1 /**
  2  * Creates an instance of pdqmus.KeySignature
  3  * An key signature musical notation. 
  4  *
  5  * @constructor
  6  * @this {pdqmus.KeySignature}
  7  * @param {number} onset Onset time, in seconds
  8  * @param {number} fifths number of clicks on the circle of fifths in the key signature, e.g. 1 is 1 sharp, -1 is 1 flat 
  9  * @param {string} key signature mode, e.g. pdqmus.KeySignature.MODE_MAJOR
 10  * @param {number} onsetMetric Metric onset, in beats
 11  */
 12 
 13 
 14 pdqmus.KeySignature = function(onset, fifths, mode, onsetMetric)
 15 {
 16     this.onset = onset;
 17     this.fifths = fifths;
 18     this.mode = mode;
 19     this.duration = 0;
 20 }
 21 
 22 pdqmus.KeySignature.prototype = new pdqmus.Notation;
 23 
 24 pdqmus.KeySignature.prototype.clone = function()
 25 {
 26     return new pdqmus.KeySignature(
 27         this.onset,
 28         this.fifths,
 29         this.mode,
 30         this.onsetMetric
 31         );
 32 };
 33 
 34 pdqmus.KeySignature.MODE_MAJOR = "major";
 35 pdqmus.KeySignature.MODE_MINOR = "minor";
