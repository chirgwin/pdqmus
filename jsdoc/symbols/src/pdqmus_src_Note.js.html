1 /**
  2  * Creates an instance of pdqmus.Note, a musical note object
  3  *
  4  * @constructor
  5  * @this {pdqmus.Note}
  6  * @param {number} onset Onset time, in seconds
  7  * @param {number} duration Duration time, in seconds
  8  * @param {number} noteNum MIDI note number 0..127
  9  * @param {number} velocity MIDI velocity 0..127
 10  * @param {boolean} isRest True if note is a rest
 11  * @param {number} onsetMetric Metric onset, in beats
 12  * @param {number} durationMetric Metric duration, in beats, e.g. 1.0 is a quarter note
 13  */
 14 
 15 pdqmus.Note = function(onset, duration, noteNum, velocity, isRest,
 16                        onsetMetric, durationMetric)
 17 {
 18     this.onset = onset;
 19     this.duration = duration;
 20     this.noteNum = noteNum;
 21     this.velocity = velocity || 60;
 22     this.onsetMetric = onsetMetric;
 23     this.durationMetric = durationMetric;
 24     this.isRest = isRest || false;
 25 }

 39 }

 41 //scale from 0..127 to 0..1
 42 pdqmus.Note.midiVelocityToAmplitude  = function(midiVelocity)
 43 {
 44     return 1 / 127 * midiVelocity;
 45 };
 46 
 47 pdqmus.Note.midiNoteFromFrequency = function(frequency)
 48 {
 49     return 12 * Math.log2(frequency / 440) + 69;
 50 };
 51 
 52 pdqmus.Note.frequencyFromMidiNote = function(midiNote)
 53 {
 54     //return 440 * Math.pow(2, (midiNote - 69) / 12);
 55     return (pdqmus.Note.CONCERT_A_FREQUENCY / 32) * Math.pow(2, (midiNote - 9) / 12);
 56     //or use lookup table
 57     //return pdqmus.Note.MidiFrequencies[midiNote];    
 58 };

 62 pdqmus.Note.randomNotes = function randomNotes(numNotes, isSoft)
 63 {    
 64     var notes = [];
 65     var onset = 0;
 66     for (var i = 0; i < numNotes; i++)
 67     {
 68         // note number 0..127
 69         var randNote = Math.floor(Math.random() * 100) + 10;
 70         // velocity 0..127
 71         if (isSoft) 
 72         {
 73             var randVelocity = Math.floor(Math.random() * 40) + 10;        
 74         }
 75         else 
 76         {
 77             var randVelocity = Math.floor(Math.random() * 50) + 67;
 78         }
 79         // duration, in seconds
 80         var randDuration = Math.random();
 81         var adjustedDuration = randDuration * 0.9;
 82         var note = new pdqmus.Note(onset, adjustedDuration, randNote, randVelocity);
 83         onset += randDuration;
 84         notes.push(note);
 85     }
 86     return notes;
 87 };
 88 
 89 pdqmus.Note.chromaticSequence = function (start, end, duration)
 90 {    
 91     var onset = 0;
 92     var notes = [];  
 93     for (var i = start; i <= end; i++)
 94     {
 95         var note = new pdqmus.Note(onset, duration, i, 60);
 96         notes.push(note);
 97         onset += duration;
 98     }
 99     return notes;
100 };
