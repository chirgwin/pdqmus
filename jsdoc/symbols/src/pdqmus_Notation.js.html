1 /**
  2  * Creates an instance of pdqmus.Notation, an abstract musical notation object
  3  *
  4  * @constructor
  5  * @this {pdqmus.Notation}
  6  * @param {number} onset Onset time, in seconds
  7  * @param {number} duration Duration time, in seconds
  8  * @param {number} onsetMetric Metric onset, in beats
  9  * @param {number} durationMetric Metric duration, in beats, e.g. 1.0 is a quarter note
 10  */
 11 pdqmus.Notation = function(onset, duration, onsetMetric, durationMetric)
 12 {
 13     this.onset = onset;
 14     this.duration = duration;
 15     this.onsetMetric = onsetMetric;
 16     this.durationMetric = durationMetric;
 17 };
 18 
 19 pdqmus.Notation.prototype.durationFromMetric = function(tempo)
 20 {
 21     this.duration = 60 / tempo * this.durationMetric;
 22 };
 23 
 24 pdqmus.Notation.prototype.onsetFromMetric = function(tempo)
 25 {
 26     this.onset = 60 / tempo * this.onsetMetric;
 27 };
 28 
 29 pdqmus.Notation.prototype.convertFromMetric = function(tempo)
 30 {
 31     this.durationFromMetric(tempo);
 32     this.onsetFromMetric(tempo);
 33 };
 34 
 35 pdqmus.Notation.prototype.getNearestNotelistDur = function(bpm)
 36 {
 37     if (!bpm)
 38     {
 39         bpm = 60;
 40     }
 41     //TODO: handle dots
 42     var nearestDurationIdx = 0;
 43     var nearestDistance = 9999;
 44     for (var i = 0; i < pdqmus.Note.METRIC_DURATIONS.length; i++)
 45     {
 46         var adjustedDuration = this.duration * (bpm / 60);
 47         var distance = Math.abs(pdqmus.Note.METRIC_DURATIONS[i] - adjustedDuration);
 48         
 49         if (distance < nearestDistance)
 50         {
 51             nearestDistance = distance;
 52             nearestDurationIdx = i;            
 53         }
 54     }  
 55     
 56     return nearestDurationIdx;
 57 };

 59 pdqmus.Notation.sortByDurationDesc = function(a, b)
 60 {
 61     if(a.duration > b.duration)
 62     {
 63         return -1;
 64     }
 65     else if(a.duration < b.duration)
 66     {
 67         return 1;
 68     }
 69     return 0;
 70 };
 71 
 72 pdqmus.Notation.sortByOnset = function(a, b)
 73 {
 74     if(a.onset < b.onset )
 75     {
 76         return -1;
 77     }
 78     else if( a.onset > b.onset)
 79     {
 80         return 1;
 81     }
 82     return 0;
 83 };

 86 /*
 87  * generate a Nightingale notelist from a series of notes
 88  * assumes single staff, single part, single voice
 89  */
 90 pdqmus.Notation.generateNotelist = function(notes, bpm)
 91 {
 92     var time = 0;
 93     if (!bpm)
 94     {
 95         bpm = 60;
 96     }
 97     var notelist = "%%Notelist-V2 file='exported from pdqm.us'  partstaves=1 0 startmeas=1" + pdqmus.Util.NEWLINE
 98     + "C stf=1 type=3" + pdqmus.Util.NEWLINE  + "K stf=1 KS=0 #" + pdqmus.Util.NEWLINE
 99     + "M stf=1 '' q=" + bpm + pdqmus.Util.NEWLINE;
100     
101     for (var i = 0; i < notes.length; i++)
102     {    
103         var note = notes[i];
104         var duration = Math.round(note.duration * pdqmus.Note.NOTELIST_TICKS_PER_QUARTER);
105         notelist += "N t=" + time + " v=1 npt=1 stf=1 dur=" + note.getNearestNotelistDur(bpm) 
106         + " dots=0 nn=" + note.noteNum
107         + " acc=0 eAcc=0 pDur=" + (duration * 0.9)
108         +  " vel=75 ...... appear=1" + pdqmus.Util.NEWLINE;        
109         time += duration;
110     }
111     
112     //tack a barline on the end
113     notelist +=  "/ t=" + time + " type=1" + pdqmus.Util.NEWLINE;
114     
115     return notelist;    
116 };
