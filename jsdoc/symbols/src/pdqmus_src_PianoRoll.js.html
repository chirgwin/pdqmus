1 /**
  2  * Creates an instance of pdqmus.PianoRoll,
  3  * A piano roll is a player piano roll style visual representation.
  4  *
  5  * @constructor
  6  * @this {pdqmus.PianoRoll}
  7  
  8  notes, unitWidth, unitHeight, cursorColor, pitchRange, destroyOnFinish
  9  
 10  * @param {Array} notes An array of pdqmus.Note objects
 11  * @param {number} unitWidth width of each time unit, in pixels
 12  * @param {number} unitHeight height of each pitch unit, in pixels
 13  * @param {string} cursorColor CSS color string of the cursor color
 14  * @param {number} pitchRange range of pitches, defaults to 127
 15  * @param {boolean} destroyOnFinish whether or not to remove the paino roll from the DOM when playback is complete 
 16  */
 17 pdqmus.PianoRoll = function(notes, unitWidth, unitHeight,
 18                             cursorColor, pitchRange, destroyOnFinish)
 19 {
 20     this.notes = notes;
 21     
 22     DEFAULT_PITCH_RANGE = 127;
 23     DEFAULT_UNIT_WIDTH = 10;
 24     DEFAULT_UNIT_HEIGHT = 2;
 25     DEFAULT_COLOR = "#36c";
 26     DEFAULT_CURSOR_COLOR = "rgba(200, 100, 50, 0.04)";
 27    
 28     var _cursorPosition = 0;
 29     var _totalLength = 0;
 30     var _canvasRef = document.createElement("canvas");;
 31     var _context = _canvasRef.getContext("2d");
 32 
 33     //corresponds to seconds per pixel
 34     var _unitWidth = unitWidth || DEFAULT_UNIT_WIDTH;
 35     var _unitHeight = unitHeight || DEFAULT_UNIT_HEIGHT;    
 36     var _cursorColor = cursorColor || DEFAULT_CURSOR_COLOR;
 37     var _pitchRange = pitchRange || DEFAULT_PITCH_RANGE;
 38     var _destroyOnFinish = destroyOnFinish || false;
 39     self = this;
 40     this.render = function(color)
 41     {
 42         var lastNote = this.notes[this.notes.length - 1];
 43         _totalLength = (lastNote.onset + lastNote.duration) * _unitWidth;
 44         _canvasRef.width = _totalLength;
 45         _canvasRef.height = _unitHeight * _pitchRange;
 46         _context.strokeStyle = DEFAULT_COLOR;
 47         if (color) 
 48         {
 49             _context.strokeStyle = color;
 50         }
 51         _context.lineWidth = _unitHeight;
 52         _context.beginPath();
 53         for (var i = 0; i < this.notes.length; i ++)
 54         {
 55             var note = this.notes[i];
 56             var x = note.onset * _unitWidth;
 57             
 58             var y = _canvasRef.height - (note.noteNum * _unitHeight);
 59             var length = note.duration * _unitWidth;
 60             _context.moveTo(x, y);
 61             _context.lineTo(x + length, y);
 62             _context.stroke();
 63         }
 64         _context.save();
 65         return _canvasRef;
 66     }    
 67     
 68     this.play = function()
 69     {
 70         renderCursor();
 71     }
 72     
 73     function destroy()
 74     {
 75        _canvasRef.parentNode.removeChild(_canvasRef);    
 76     }
 77 
 78     /* TODO: fix this to render a proper cursor
 79      * 
 80     */
 81     function renderCursor()
 82     {
 83         _context.strokeStyle = _cursorColor;
 84         _context.moveTo(_cursorPosition, 0);
 85         _context.lineTo(_cursorPosition, _canvasRef.height);    
 86         _context.stroke();
 87         _cursorPosition ++;
 88         if (_cursorPosition <= _totalLength) 
 89         {
 90             setTimeout(renderCursor, 1000 / (_unitWidth));
 91         }
 92         else
 93         {
 94             if (_destroyOnFinish)
 95             {
 96                 destroy();
 97             }
 98             _context.restore();
 99         }
100     }    
101 }
102 
103 /* convenience function for lazy drawing */
104 pdqmus.PianoRoll.prototype.draw = function()
105 {
106     var canvas = this.render();
107     canvas.style.border = "1px solid #352";
108     canvas.style.padding = "10px";
109     document.body.appendChild(canvas);   
110 }
