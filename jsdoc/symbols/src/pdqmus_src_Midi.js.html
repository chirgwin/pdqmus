1 /**
  2  * Creates an instance of pdqmus.Midi, a file-like MIDI object 
  3  *
  4  * @constructor
  5  * @this {pdqmus.Midi}
  6  * @param {number} numTracks Number of tracks in the midi "file"
  7  * @param {number} tempo  a integer indicating the tempo of the MIDI "file", in beats per minute.
  8  */
  9 
 10 pdqmus.Midi = function(numTracks, tempo)
 11 {
 12     const HEADER_CHUNK = "MThd";
 13     const TRACK_CHUNK = "MTrk";   
 14     const SINGLE_TRACK_FORMAT = 0;
 15     const SIMULTANEOUS_TRACKS_FORMAT = 1;
 16     const SEPARATE_TRACKS_FORMAT = 2;
 17     const STATUS = 0xFF;
 18     const SEQUENCE_NUMBER = 0x00; 
 19     const TEXT = 0x01;
 20     const COPYRIGHT = 0x02;
 21     const TRACK_NAME = 0x03;
 22     const INSTRUMENT = 0x04;
 23     const LYRIC = 0x05;
 24     const MARKER = 0x06;
 25     const CUE_POINT = 0x07;
 26     const PATCH_NAME = 0x08;
 27     const PORT_NAME = 0x09;
 28     const CHANNEL_PREFIX = 0x20;
 29     const END_OF_TRACK = 0x2F;
 30     const TEMPO = 0x51;
 31     const SMPTE_OFFSET = 0x54;
 32     const TIME_SIGNATURE = 0x58;
 33     const KEY_SIGNATURE = 0x59;
 34     const PROPRIETARY = 0x7F;
 35     const PULSE = 24;
 36     const THIRTYSECONDS_PER_PULSE = 8;    
 37     const MICROSECONDS_PER_MINUTE = 60000000;
 38     const HEADER_LENGTH = 6;        
 39     const DIVISIONS = 480;      
 40 	    
 41     const MAX_SOUNDING_TRACKS = 24;
 42     const MAX_CHANNELS = 16;  
 43 
 44     const NOTE_OFF = 0x8;
 45     const NOTE_ON = 0x9;
 46     const AFTERTOUCH = 0xA;
 47     const CONTROL_CHANGE = 0xB;
 48     const PROGRAM_CHANGE = 0xC;
 49     const CHANNEL_PRESSURE = 0xD;
 50     const PITCH_WHEEL = 0xE;    
 51     const SYSTEM = 0xF;
 52     const SYSTEM_EX = 0xF0;
 53     const MTC_QUARTER_FRAME = 0xF1;
 54     const SONG_POSITION_POINTER = 0xF2;
 55     const SONG_SELECT = 0xF3;
 56     const TUNE_REQUEST = 0xF6;
 57     const CLOCK = 0xF8;
 58     const TICK = 0xF9;
 59     const START = 0xFA;
 60     const CONTINUE = 0xFB;
 61     const STOP = 0xFC;
 62     const ACTIVE_SENSE = 0xFE;
 63     const RESET = 0xFF;
 64     const SYSTEM_EX_END = 0xF7;
 65     const MIDI_MIDDLE_C = 60;    
 66     const KEY_SIGNATURE_MODE_MAJOR = 0;
 67     const KEY_SIGNATURE_MODE_MINOR = 1;
 68 	
 69     const DEFAULT_NUM_TRACKS = 1;
 70     const DEFAULT_TEMPO = 120;
 71     const DEFAULT_INSTRUMENT_NAME = "Contrabass Ukulelephonium";
 72     const DEFAULT_CHANNEL = 0;
 73     const DEFAULT_PATCH_NUMBER = 0;
 74     const DEFAULT_TITLE = "MIDI Created by Javascript";
 75     const DEFAULT_COPYRIGHT = "Copyleft, Music Hack Day Boston 2010 rules";
 76 	
 77     var _bpm = tempo || DEFAULT_TEMPO;
 78     var _header = "";
 79     var _data = "";
 80     
 81     var _numTracks = numTracks || DEFAULT_NUM_TRACKS;
 82 
 83 
 84     function init()
 85     {
 86         createHeader();	
 87         createTempoTrack();	
 88     }
 89 
 90     function createHeader ()
 91     {
 92 	_header = HEADER_CHUNK
 93         + pdqmus.Util.strFromDword(HEADER_LENGTH, pdqmus.Util.BIG_ENDIAN)
 94         + pdqmus.Util.strFromWord(SIMULTANEOUS_TRACKS_FORMAT, pdqmus.Util.BIG_ENDIAN)
 95         + pdqmus.Util.strFromWord(_numTracks + 1, pdqmus.Util.BIG_ENDIAN)
 96         + pdqmus.Util.strFromWord(DIVISIONS, pdqmus.Util.BIG_ENDIAN);
 97     }
 98 	
 99     function writeMessages(messages)
100     {
101         var prevTime = 0;
102         var currentTime = 0;
103         var messagesStr = ""
104         for (var i = 0; i < messages.length; i++)
105         {
106             currentTime = Math.round((messages[i].onset * _bpm / 60) * DIVISIONS);
107             messagesStr += pdqmus.Util.strFromVariableInt(currentTime - prevTime);
108             messagesStr += messages[i].write();
109             prevTime = currentTime;
110         }   
111         return messagesStr;        
112     }
113       
114     // eventually may want to make this public so that 
115     // caller may add things to this track
116     function createTempoTrack()
117     {			
118         //hardwire a title and a copyright
119         var title = DEFAULT_TITLE;
120         var copyright = DEFAULT_COPYRIGHT;
121 		
122         var tempo = Math.round(MICROSECONDS_PER_MINUTE / _bpm);
123 
124         var track = pdqmus.Util.strFromByte(0)
125         + pdqmus.Util.strFromByte(STATUS)
126         + pdqmus.Util.strFromByte(TEMPO)
127         //chunk size
128         + pdqmus.Util.strFromByte(3)
129         + pdqmus.Util.strFromByte(tempo >> 16)
130         + pdqmus.Util.strFromByte(tempo >> 8)
131         + pdqmus.Util.strFromByte(tempo)
132 
133         + pdqmus.Util.strFromByte(0)
134         + pdqmus.Util.strFromByte(STATUS)
135         + pdqmus.Util.strFromByte(TRACK_NAME)
136         + pdqmus.Util.strFromByte(title.length)
137         + title		
138         + pdqmus.Util.strFromByte(0)
139         + pdqmus.Util.strFromByte(STATUS)
140         + pdqmus.Util.strFromByte(COPYRIGHT)
141         + pdqmus.Util.strFromByte(copyright.length)
142         + copyright;
143 		
144         //hardwire a key signature and time signature
145         messages = [];		
146         messages.push(new pdqmus.MidiKeySignature(0,0,0));
147         messages.push(new pdqmus.MidiTimeSignature(1,0,4,4));
148         track += writeMessages(messages)
149             + pdqmus.Util.strFromByte(0)
150             + pdqmus.Util.strFromByte(STATUS)
151             + pdqmus.Util.strFromByte(END_OF_TRACK)
152             + pdqmus.Util.strFromByte(0);
153         
154         _data += TRACK_CHUNK
155         + pdqmus.Util.strFromDword(track.length, pdqmus.Util.BIG_ENDIAN)
156         + track;
157     }
158     
159     this.addTrack = function (elements, channel, GMInstrumentNum)
160     {
161 		//hardwire instrument name
162 		var instrumentName = DEFAULT_INSTRUMENT_NAME;
163 		if (!(channel >= 0))
164 		{
165 			channel = DEFAULT_CHANNEL;
166 		}
167         if (!(GMInstrumentNum >= 0))
168         {
169             channel = DEFAULT_PATCH_NUMBER;
170         }
171 		 
172         var track = pdqmus.Util.strFromByte(0)
173         + pdqmus.Util.strFromByte(PROGRAM_CHANGE * 0x10 + channel)
174         + pdqmus.Util.strFromByte(GMInstrumentNum)
175 		
176         + pdqmus.Util.strFromByte(0)
177         + pdqmus.Util.strFromByte(STATUS)
178         + pdqmus.Util.strFromByte(INSTRUMENT)
179         + pdqmus.Util.strFromByte(instrumentName.length)
180 		+ instrumentName;
181 			        
182         var messages = [];
183         for (var i = 0; i < elements.length; i++)
184         {
185 			var note = elements[i];
186             messages.push(new pdqmus.MidiNote(i, note.onset, NOTE_ON, channel,
187                                        note.noteNum, note.velocity)); 
188             //note off, zero velocity
189             messages.push(new pdqmus.MidiNote(i, note.onset + note.duration, NOTE_OFF, channel,
190                                        note.noteNum, 0));
191         }
192         
193         //sort messages
194         messages.sort(pdqmus.Notation.sortByOnset);
195         
196         track += writeMessages(messages)        
197         //pad the end a little
198         + pdqmus.Util.strFromVariableInt(Math.round(DIVISIONS / 10))
199 		//end of track...
200         + pdqmus.Util.strFromByte(STATUS)
201         + pdqmus.Util.strFromByte(END_OF_TRACK)
202         + pdqmus.Util.strFromByte(0);
203         
204         _data += TRACK_CHUNK
205         + pdqmus.Util.strFromDword(track.length, pdqmus.Util.BIG_ENDIAN)
206         + track;
207     }
208 	  	
209 	this.getDataUrl = function()
210 	{
211 		return "data:" + pdqmus.Midi.MIME_TYPE + ";base64," + btoa(_header + _data); 
212 	}
213 	
214 	init();	
215 }
216 
217 pdqmus.Midi.prototype.play = function()
218 {
219 	var ms = pdqmus.MidiSound.createFromUrl(this.getDataUrl());
220 	ms.play();
221 }

223 pdqmus.MidiMessage = function(index, onset, statusCode)
224 {
225     this.index = index;
226     this.onset = onset;
227     this.statusCode = statusCode;
228 }        
229 
230 pdqmus.MidiNote = function(index, onset, statusCode, channel, noteNum, velocity)
231 {
232     this.index = index;
233     this.onset = onset;
234     this.statusCode = statusCode;
235     this.channel = channel;
236     this.noteNum = noteNum;
237     this.velocity = velocity;
238 }
239 
240 pdqmus.MidiNote.prototype = new pdqmus.MidiMessage;
241 
242 pdqmus.MidiNote.prototype.write = function()
243 {
244     return pdqmus.Util.strFromByte(this.statusCode * 0x10 + this.channel, pdqmus.Util.BIG_ENDIAN)
245            + pdqmus.Util.strFromByte(this.noteNum, pdqmus.Util.BIG_ENDIAN)
246            + pdqmus.Util.strFromByte(this.velocity, pdqmus.Util.BIG_ENDIAN);
247 }
248 
249 pdqmus.MidiKeySignature = function(index, onset, fifths)
250 {
251     this.index = index;
252     this.onset = onset;
253     //always key signature status code
254     this.statusCode = 0x59;
255     this.fifths = fifths;
256     //always default to "major" mode
257     this.mode = 0;
258 }
259 
260 pdqmus.MidiKeySignature.prototype = new pdqmus.MidiMessage;
261 pdqmus.MidiKeySignature.prototype.write = function()
262 {
263     return pdqmus.Util.strFromByte(0xFF)
264            + pdqmus.Util.strFromByte(this.statusCode)	
265             //chunk size
266            + pdqmus.Util.strFromByte(2)	
267            + pdqmus.Util.strFromByte(this.fifths)
268            + pdqmus.Util.strFromByte(this.mode);
269 }
270 
271 
272 pdqmus.MidiTimeSignature = function(index, onset, numerator, denominator)
273 {
274     this.index = index;
275     this.onset = onset;
276     //always time signature status code
277     this.statusCode = 0x58;
278     this.numerator = numerator;
279     this.denominator = denominator;
280     this.pulse = 24;
281     this.thirtySecondsPerPulse = 8;
282 }
283 
284 pdqmus.MidiTimeSignature.prototype = new pdqmus.MidiMessage;
285 pdqmus.MidiTimeSignature.prototype.write = function()
286 {
287     //"meta" status
288     return pdqmus.Util.strFromByte(0xFF, pdqmus.Util.BIG_ENDIAN)
289            + pdqmus.Util.strFromByte(this.statusCode)	
290             //chunk size
291            + pdqmus.Util.strFromByte(4)	
292            + pdqmus.Util.strFromByte(this.numerator)
293             //denominator is log2 of normal music notation denominator
294            + pdqmus.Util.strFromByte(Math.round(Math.log(this.denominator) / Math.LN2))
295            + pdqmus.Util.strFromByte(this.pulse)
296            + pdqmus.Util.strFromByte(this.thirtySecondsPerPulse);
297 }
