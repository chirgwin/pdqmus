1 /**
  2  * Creates an instance of pdqmus.Sequencer,
  3  * A sequencer is a series of sounds that can be rendered as a grid
  4  *
  5  * @constructor
  6  * @this {pdqmus.Sequencer}
  7  * @param {Array} sounds An array of pdqmus.Sound objects
  8  * @param {number} tempo beats per minute, integer
  9  * @param {string} meterType type of meter, e.g. simple (pdqmus.Sequencer.SIMPLE_METER) or compound
 10  * @param {number} numBigBeats number of big beats, defaults to 4
 11  */
 12 
 13 pdqmus.Sequencer = function(sounds, tempo, meterType, numBigBeats)
 14 {
 15     const DEFAULT_TEMPO = 120;
 16     //TODO: make this an array of actual sample objects, instead of wav files
 17     
 18     const DEFAULT_METER_TYPE = pdqmus.Sequencer.SIMPLE_METER;
 19     
 20     const DEFAULT_NUM_BIG_BEATS = 4;
 21     
 22     var _sounds = sounds;
 23     //BPM
 24     //TODO: change tempo on existing instance?
 25     this.tempo = tempo || DEFAULT_TEMPO;
 26     this.meterType = meterType || DEFAULT_METER_TYPE;
 27     this.numBigBeats = numBigBeats || DEFAULT_NUM_BIG_BEATS;
 28     
 29     //milliseconds per note division
 30     var loopDuration = 60000 / this.tempo  / this.meterType;
 31 
 32     //beats by voice grid    
 33     var numVoices = _sounds.length;
 34     var numBeats = this.numBigBeats * this.meterType;
 35 
 36     var currentBeatIdx = 0;
 37     
 38     this.isPlaying = false;
 39     this.isLooping = true;
 40     var _loopTimeout;
 41     //TODO: try to get this to work
 42     this.allowSampleOverlap = false;
 43     
 44     var self = this;
 45     function init()
 46     {
 47         //set up sounds
 48         var _sounds2d = [];
 49         for (var i = 0; i < numVoices; i++) 
 50         {
 51             _sounds2d[i] = [];
 52             _sounds2d[i][0] = _sounds[i];
 53             //clone sounds into each beat slot to allow overlap
 54             if (self.allowSampleOverlap) 
 55             {
 56                 for (var j = 1; j < numBeats; j++) 
 57                 {
 58                     _sounds2d[i][j] = _sounds[i].clone();
 59                     console.log(i, j, _sounds2d[i][j].id)
 60                 }
 61             }
 62         }    
 63         _sounds = _sounds2d;        
 64     }
 65     //loop    
 66     this.loop = function()
 67     {    
 68     
 69         if (currentBeatIdx >= numBeats)
 70         {
 71             currentBeatIdx = 0;
 72             if (!self.isLooping)
 73             {
 74                 return;                
 75             }
 76         }        
 77 
 78         for (var i = 0; i < numVoices; i ++)
 79         {
 80             for (var j = 0; j < numBeats; j ++)
 81             {
 82                 var element = document.getElementById(i + "_" + j);
 83                 //TODO: keep scroll in view
 84                 //element.scrollIntoView();          
 85                 //console.log(Util.getObjectPosition(element).x);
 86                 
 87                 element.className = "beatbox";
 88                 if (element.isDownbeat)
 89                 {
 90                     element.className += " downbeat";
 91                 }
 92                 if (element.beatOn)
 93                 {
 94                     element.className += " beatOn";
 95                 }
 96                 else
 97                 {
 98                     element.className += " beatOff";                
 99                 }
100                 if (j == currentBeatIdx)
101                 {
102                     element.className += " beatHighlighted";                
103                 }
104             }
105         }
106 
107         
108         for (var i = 0; i < _sounds.length; i++)
109         {
110             if (document.getElementById(i + "_" + currentBeatIdx).beatOn)
111             {
112                 if (!self.allowSampleOverlap) 
113                 {
114                     //? _sounds[i][0].pause()
115                     _sounds[i][0].play();
116                 }
117                 else
118                 {
119                     _sounds[i][currentBeatIdx].play();                    
120                 }
121             }
122         }
123 
124         currentBeatIdx ++;
125 
126         if (self.isPlaying)
127         {
128             _loopTimeout = setTimeout(self.loop, loopDuration);
129         }
130         
131     }
132 
133 
134     function handleMouseDown(event)
135     {
136         event.target.beatOn = !event.target.beatOn;
137         
138         if (event.target.beatOn)
139         {
140             event.target.className = "beatbox beatOn";
141         }
142         else
143         {
144             event.target.className = "beatbox beatOff";            
145         }
146         
147         if (event.target.isDownbeat)
148         {
149             event.target.className += " downbeat";            
150         }
151         
152     }
153     
154     this.render = function(container)
155     {        
156         for (var i = 0; i < numVoices; i ++)
157         {
158             var row = document.createElement("div");
159             for (var j = 0; j < numBeats; j ++)
160             {
161                 var inputRef = document.createElement("div");
162                 inputRef.setAttribute("id", i + "_" + j); 
163                 inputRef.beatOn = false;
164                 
165                 inputRef.addEventListener("mousedown", handleMouseDown, "true");               
166                 if (j > 0 && j % this.meterType == 0)
167                 {            
168                     inputRef.isDownbeat = true;
169                     inputRef.className = "beatbox beatOff downbeat";
170                 }
171                 else
172                 {
173                     inputRef.className = "beatbox beatOff";                
174                 }
175                 row.appendChild(inputRef);
176                 row.className = "row";
177             }
178             container.appendChild(row);
179         }  
180     }
181     
182     this.pauseAllSounds = function()
183     {    
184        //this.sounds
185     }
186     this.play = function()
187     {
188         this.isPlaying = true;
189         self.loop();
190     }
191 
192     this.pause = function()
193     {
194         self.isPlaying = false;
195         if (_loopTimeout)
196         {
197             clearTimeout(_loopTimeout);
198         }
199         self.pauseAllSounds();
200     }
201 
202     init();
203 }

205 //convenience method for lazy drawing
206 //TODO: refactor render method to return canvas
207 pdqmus.Sequencer.prototype.draw = function()
208 {
209     var sDiv = document.createElement("div");
210     sDiv.className = "sequencer";
211     this.render(sDiv);
212     document.body.appendChild(sDiv);
213 }

216 pdqmus.Sequencer.DUPLE_METER = 2;
217 pdqmus.Sequencer.SINGLE_METER = 1;
