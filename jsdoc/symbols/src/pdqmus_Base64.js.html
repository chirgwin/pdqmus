  6 var Base64 = {};  // Base64 namespace
  7 
  8 Base64.code = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/=";

 10 /**
 11  * Encode string into Base64, as defined by RFC 4648 [http://tools.ietf.org/html/rfc4648]
 12  * (instance method extending String object). As per RFC 4648, no newlines are added.
 13  *
 14  * @param {String} str The string to be encoded as base-64
 15  * @param {Boolean} [utf8encode=false] Flag to indicate whether str is Unicode string to be encoded 
 16  *   to UTF8 before conversion to base64; otherwise string is assumed to be 8-bit characters
 17  * @returns {String} Base64-encoded string
 18  */ 
 19 Base64.encode = function(str, utf8encode) {  // http://tools.ietf.org/html/rfc4648
 20   utf8encode =  (typeof utf8encode == 'undefined') ? false : utf8encode;
 21   var o1, o2, o3, bits, h1, h2, h3, h4, e=[], pad = '', c, plain, coded;
 22   var b64 = Base64.code;
 23    
 24   plain = utf8encode ? Utf8.encode(str) : str;
 25   
 26   c = plain.length % 3;  // pad string to length of multiple of 3
 27   if (c > 0) { while (c++ < 3) { pad += '='; plain += '\0'; } }
 28   // note: doing padding here saves us doing special-case packing for trailing 1 or 2 chars
 29    
 30   for (c=0; c<plain.length; c+=3) {  // pack three octets into four hexets
 31     o1 = plain.charCodeAt(c);
 32     o2 = plain.charCodeAt(c+1);
 33     o3 = plain.charCodeAt(c+2);
 34       
 35     bits = o1<<16 | o2<<8 | o3;
 36       
 37     h1 = bits>>18 & 0x3f;
 38     h2 = bits>>12 & 0x3f;
 39     h3 = bits>>6 & 0x3f;
 40     h4 = bits & 0x3f;
 41 
 42     // use hextets to index into code string
 43     e[c/3] = b64.charAt(h1) + b64.charAt(h2) + b64.charAt(h3) + b64.charAt(h4);
 44   }
 45   coded = e.join('');  // join() is far faster than repeated string concatenation in IE
 46   
 47   // replace 'A's from padded nulls with '='s
 48   coded = coded.slice(0, coded.length-pad.length) + pad;
 49    
 50   return coded;
 51 }

 53 /**
 54  * Decode string from Base64, as defined by RFC 4648 [http://tools.ietf.org/html/rfc4648]
 55  * (instance method extending String object). As per RFC 4648, newlines are not catered for.
 56  *
 57  * @param {String} str The string to be decoded from base-64
 58  * @param {Boolean} [utf8decode=false] Flag to indicate whether str is Unicode string to be decoded 
 59  *   from UTF8 after conversion from base64
 60  * @returns {String} decoded string
 61  */ 
 62 Base64.decode = function(str, utf8decode) {
 63   utf8decode =  (typeof utf8decode == 'undefined') ? false : utf8decode;
 64   var o1, o2, o3, h1, h2, h3, h4, bits, d=[], plain, coded;
 65   var b64 = Base64.code;
 66 
 67   coded = utf8decode ? Utf8.decode(str) : str;
 68   
 69   
 70   for (var c=0; c<coded.length; c+=4) {  // unpack four hexets into three octets
 71     h1 = b64.indexOf(coded.charAt(c));
 72     h2 = b64.indexOf(coded.charAt(c+1));
 73     h3 = b64.indexOf(coded.charAt(c+2));
 74     h4 = b64.indexOf(coded.charAt(c+3));
 75       
 76     bits = h1<<18 | h2<<12 | h3<<6 | h4;
 77       
 78     o1 = bits>>>16 & 0xff;
 79     o2 = bits>>>8 & 0xff;
 80     o3 = bits & 0xff;
 81     
 82     d[c/4] = String.fromCharCode(o1, o2, o3);
 83     // check for padding
 84     if (h4 == 0x40) d[c/4] = String.fromCharCode(o1, o2);
 85     if (h3 == 0x40) d[c/4] = String.fromCharCode(o1);
 86   }
 87   plain = d.join('');  // join() is far faster than repeated string concatenation in IE
 88    
 89   return utf8decode ? Utf8.decode(plain) : plain; 
 90 }

 94 /*  Utf8 class: encode / decode between multi-byte Unicode characters and UTF-8 multiple          */
 95 /*              single-byte character encoding (c) Chris Veness 2002-2010                         */
 96 /* - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -  */
 97 
 98 var Utf8 = {};  // Utf8 namespace
 99 
100 /**
101  * Encode multi-byte Unicode string into utf-8 multiple single-byte characters 
102  * (BMP / basic multilingual plane only)
103  *
104  * Chars in range U+0080 - U+07FF are encoded in 2 chars, U+0800 - U+FFFF in 3 chars
105  *
106  * @param {String} strUni Unicode string to be encoded as UTF-8
107  * @returns {String} encoded string
108  */
109 Utf8.encode = function(strUni) {
110   // use regular expressions & String.replace callback function for better efficiency 
111   // than procedural approaches
112   var strUtf = strUni.replace(
113       /[\u0080-\u07ff]/g,  // U+0080 - U+07FF => 2 bytes 110yyyyy, 10zzzzzz
114       function(c) { 
115         var cc = c.charCodeAt(0);
116         return String.fromCharCode(0xc0 | cc>>6, 0x80 | cc&0x3f); }
117     );
118   strUtf = strUtf.replace(
119       /[\u0800-\uffff]/g,  // U+0800 - U+FFFF => 3 bytes 1110xxxx, 10yyyyyy, 10zzzzzz
120       function(c) { 
121         var cc = c.charCodeAt(0); 
122         return String.fromCharCode(0xe0 | cc>>12, 0x80 | cc>>6&0x3F, 0x80 | cc&0x3f); }
123     );
124   return strUtf;
125 }
126 
127 /**
128  * Decode utf-8 encoded string back into multi-byte Unicode characters
129  *
130  * @param {String} strUtf UTF-8 string to be decoded back to Unicode
131  * @returns {String} decoded string
132  */
133 Utf8.decode = function(strUtf) {
134   var strUni = strUtf.replace(
135       /[\u00c0-\u00df][\u0080-\u00bf]/g,                 // 2-byte chars
136       function(c) {  // (note parentheses for precence)
137         var cc = (c.charCodeAt(0)&0x1f)<<6 | c.charCodeAt(1)&0x3f;
138         return String.fromCharCode(cc); }
139     );
140   strUni = strUni.replace(
141       /[\u00e0-\u00ef][\u0080-\u00bf][\u0080-\u00bf]/g,  // 3-byte chars
142       function(c) {  // (note parentheses for precence)
143         var cc = ((c.charCodeAt(0)&0x0f)<<12) | ((c.charCodeAt(1)&0x3f)<<6) | ( c.charCodeAt(2)&0x3f); 
144         return String.fromCharCode(cc); }
145     );
146   return strUni;
147 }
