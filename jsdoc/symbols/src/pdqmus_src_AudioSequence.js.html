1 /**
  2  * Creates an instance of pdqmus.AudioSequence.
  3  *
  4  * @constructor
  5  * @this {pdqmus.AudioSequence}
  6  * @param notes {Array} a sequence of pdqmus.Note objects
  7  * @param oscillatorType {string} the type of oscillator to synthesize
  8  * @param envelope {string} the type of envelope shape
  9  * @param includePath {string} path to include for WebWorker script
 10  */
 11 
 12 pdqmus.AudioSequence = function(notes, oscillatorType, envelope, includePath)
 13 {
 14     //milliseconds
 15     const INTERVAL = 10;
 16     
 17     var playhead = 0;
 18     
 19     var playLoop;
 20     var startTime;
 21     var buffer = [];
 22     var lookaheadSeconds = 2;
 23 
 24     var _notes = notes.clone();
 25     var _playbackNotes;
 26     var _oscillatorType = oscillatorType;
 27     var _envelope = envelope || null;
 28     var bufferSpent = false;
 29     var bufferStarted = false;
 30     var allNotesBuffered = false;
 31     
 32     var _aggregateWave = new pdqmus.Wave();
 33     var _aggregateWaveNeedsUpdate = true;
 34     var _startCallback = null;
 35     var _finishCallback = null;
 36 
 37     var self = this;
 38     /* set up the web worker */    
 39     var waveRenderWorker;
 40     if (!!window.Worker) 
 41     {
 42         if (includePath) 
 43         {
 44             waveRenderWorker = new Worker(includePath + "/wave_render_worker.js");
 45         }
 46         else 
 47         {
 48             waveRenderWorker = new Worker("wave_render_worker.js");
 49         }
 50 
 51         waveRenderWorker.onmessage = function(event) 
 52         {  
 53             var renderMessage = event.data;
 54             buffer.push(renderMessage);    
 55         };  
 56         
 57         waveRenderWorker.onerror = function(error) 
 58         {  
 59             //console.log("Worker error: " + error.message + "\n");
 60             throw error;  
 61         };    
 62     }    
 63 
 64     /* make a single wave "file" from all notes */
 65     function aggregateWave()
 66     {
 67         for (var i = 0; i < _notes.length; i++) 
 68         {
 69             var note = _notes[i];
 70             //TODO: add overlapping notes' samples 
 71             //otherwise, just append them
 72             _aggregateWave.render(pdqmus.Note.frequencyFromMidiNote(note.noteNum), 
 73                                   note.duration, 
 74                                   pdqmus.Note.midiVelocityToAmplitude(note.velocity), 
 75                                   oscillatorType);
 76             if (envelope)
 77             {
 78                 _aggregateWave.envelope(_envelope);
 79             }
 80         }
 81         _aggregateWaveNeedsUpdate = false;
 82     }            
 83     
 84     this.drawWaveform = function()
 85     {
 86         return self.getAggregateWave().draw();
 87     }
 88     
 89     this.getDataUrl = function()
 90     {
 91         return self.getAggregateWave().getDataUrl();
 92     }
 93 
 94     this.getAggregateWave = function()
 95     {
 96         if (_aggregateWaveNeedsUpdate)
 97         {
 98             aggregateWave();
 99         }
100         return _aggregateWave;        
101     }
102     this.pause = function()
103     {
104         throw ("unimplemented");        
105     }
106     
107     this.stop = function()
108     {
109         if (waveRenderWorker) 
110         {
111             waveRenderWorker.terminate();
112         }
113         clearTimeout(playLoop);
114         cleanUp();
115     }
116     
117     this.play = function(asynchronous)
118     {
119         //quick and dirty syncronous playback using aggregate wave
120         if (!asynchronous)
121         {
122             pdqmus.Sample.createFromUrl(self.getDataUrl()).play();
123             return;
124         }
125         //asynchronous playback
126         startTime = new Date(); 
127         _playbackNotes = _notes.clone();
128         _playbackNotes.sort(pdqmus.Notation.sortByOnset);
129         setTimeout(loop, 1);
130     }
131     
132     
133     function playNote(renderMessage)
134     {
135         pdqmus.Sample.createFromUrl(renderMessage.audioDataUrl).play();    
136     }
137     
138     this.setStartCallback = function(callback)
139     {
140         _startCallback = callback;
141         
142     }
143 
144     this.setFinishCallback = function(callback)
145     {
146         _finishCallback = callback;        
147     }
148 
149     function loop()
150     {
151         var playhead = (new Date().getTime() - startTime) / 1000;
152 
153     
154         if (buffer.length > 0 && (buffer[0].notes[0].onset <= playhead - pdqmus.AudioSequence.BUFFER_DELAY)) 
155         {
156             if (!bufferStarted) 
157             {
158                 if (_startCallback) 
159                 {
160                     _startCallback();
161                 }
162                 bufferStarted = true;                
163             }
164             if (buffer.length <= 1 && allNotesBuffered)
165             {
166                 bufferSpent = true;
167             }
168             setTimeout(playNote, 1, buffer.shift());
169         }
170         
171         if (_playbackNotes.length > 0 && _playbackNotes[0].onset <= playhead + lookaheadSeconds) 
172         {
173             var note = _playbackNotes.shift();
174             var renderNotes = [note];
175             for (var i = 1; i < _playbackNotes.length; i++)
176             {
177                 if (_playbackNotes[i].onset == note.onset)
178                 {
179                     renderNotes.push(_playbackNotes.shift());
180                 }
181                 else if (_playbackNotes[i].onset > note.onset)
182                 {
183                     break;
184                 }
185             }
186             var renderMessage = new pdqmus.RenderWaveMessage([note], _oscillatorType, _envelope);
187             waveRenderWorker.postMessage(renderMessage);
188         }
189         
190         if (_playbackNotes.length <= 0)
191         {
192             allNotesBuffered = true;
193         }
194 
195         if (!bufferSpent) 
196         {
197             playLoop = setTimeout(loop, INTERVAL);
198         }    
199         else
200         {
201             cleanUp();
202         }    
203     }
204     
205     function cleanUp()
206     {
207         if (_finishCallback) 
208         {
209             _finishCallback();
210         }
211     }
212 }
213 
214 //seconds
215 pdqmus.AudioSequence.BUFFER_DELAY = 0.3;
