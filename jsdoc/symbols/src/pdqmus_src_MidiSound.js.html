1 /**
  2  * Creates an instance of pdqmus.MidiSound,
  3  * a wrapper object for the pdqmus.Sound representation of a MIDI object 
  4  *
  5  * @constructor
  6  * @this {pdqmus.MidiSound}
  7  */
  8 pdqmus.MidiSound = function()
  9 {
 10     this.domRef = document.createElement("embed");
 11     this.parentRef = document.getElementsByTagName("body")[0];
 12 }
 13 
 14 pdqmus.MidiSound.prototype = new pdqmus.Sound;
 15 
 16 /*
 17  * public methods
 18  */
 19 
 20 pdqmus.MidiSound.prototype.play = function()
 21 {
 22     //recreate a new embed for each play
 23     //since embed doesn't have play/pause methods
 24     this.domRef.hidden = "true";
 25     //this.domRef.id = "midiEmbed" + new Date();
 26     this.domRef.autoplay = "true";
 27     this.domRef.autostart = "True";
 28     this.domRef.loop = "false";
 29     this.domRef.volume = "100%";
 30     this.domRef.style.visibility = "hidden";
 31     this.parentRef.appendChild(this.domRef);
 32 //??    this.domRef.Play();
 33 }
 34 
 35 pdqmus.MidiSound.prototype.pause = function()
 36 {
 37     this.destroy();
 38 }
 39 pdqmus.MidiSound.prototype.destroy = function()
 40 {    
 41     this.domRef.parentNode.removeChild(this.domRef);
 42     this.domRef = null;
 43 }
 44 pdqmus.MidiSound.prototype.clone = function ()
 45 {
 46     return pdqmus.MidiSound.createFromUrl(this.url);
 47 }
 48 
 49 /*
 50  * public static methods
 51  */
 52 pdqmus.MidiSound.createFromUrl = function(url)
 53 {
 54     var newMidiSound = new pdqmus.MidiSound();
 55     newMidiSound.domRef.src = url;
 56     newMidiSound.domRef.id = pdqmus.Util.guid();
 57     return newMidiSound;
 58 }
