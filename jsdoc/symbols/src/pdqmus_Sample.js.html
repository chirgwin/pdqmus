1 /**
  2  * Creates an instance of pdqmus.Sample,
  3  * A sample is an audio sample represented in the DOM.
  4  *
  5  * @constructor
  6  * @this {pdqmus.Sample}
  7  */
  8 pdqmus.Sample = function()
  9 {
 10     this.domRef;    
 11     this.domRef = document.createElement("audio");
 12     this.domRef.preload = true;
 13 }
 14 
 15 pdqmus.Sample.prototype = new pdqmus.Sound;
 16 
 17 /*
 18  * public methods
 19  */
 20 pdqmus.Sample.prototype.play = function()
 21 {
 22     this.domRef.play();
 23 }
 24 
 25 pdqmus.Sample.prototype.pause = function()
 26 {
 27     this.domRef.pause();
 28 }
 29 
 30 pdqmus.Sample.prototype.clone = function ()
 31 {
 32     return pdqmus.Sample.createFromUrl(this.url);
 33 }
 34 
 35 /*
 36  * public static methods
 37  */
 38 pdqmus.Sample.createFromUrl = function(url)
 39 {
 40     var newSample = new pdqmus.Sample();
 41     newSample.domRef.src = url;
 42     newSample.domRef.id = newSample.id = pdqmus.Util.guid();
 43     return newSample;
 44 }
